81  /**
82   @author Le Fortier 
83   
84   * This method is used to choice the map you want to play 
85   * 
86   * In the part try we have the call of the procedure in the database.
87   * cst is a CallableStatement object which call a procedure.
88   * The method prepareCall allows to call the procedure called chooseMap(?, ?) with to parameters : one INPUT parameter and one OUTPUT parameter.
89   * The method setInt allows to give the INPUT parameter.
90   * The method registerOutParameter allows to give the OUTPUT parameter which is the result of the procedure.
91   * The method execute allows to execute the procedure.
92   * After that, we write the result (the map) in a txt file.
93   * 
94   * 
95   * @param ch the choice of the user
96   * 
97   */
98      public void chooseMap(int ch) throws SQLException {
99      	try {
100         cst = conn.prepareCall("{call chooseMap(?, ?)}");
101         cst.setInt(1, ch);
102         cst.registerOutParameter(2, java.sql.Types.VARCHAR);
103         cst.execute();
104         stg = cst.getString(2);
105         		
106         File ff=new File("Map.txt");
107         FileWriter ffw=new FileWriter(ff);
108         ffw.write(stg);
109         ffw.close();
110 
111         }catch(SQLException e) {
112 	        e.printStackTrace();
113 	        System.out.println("Error");
114         }catch (IOException e) {
115         	e.printStackTrace();
116         }
